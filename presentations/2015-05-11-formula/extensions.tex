      \pgf@yb=-4\pgf@x%

      \pgf@yc=-4\pgf@x%

      \advance\pgf@yb by\pgf@y
      \advance\pgf@xc by\pgf@x
      \advance\pgf@yc by\pgf@y

      % Save centerpoint in "a" (top triangle point)
